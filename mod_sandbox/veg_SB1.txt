model{
  # Likelihood
  for(cell in 1:n) {
    for(lc in 1:7) {
      Y[cell, lc] ~ dnorm(w[cell, lc], tau[lc])
    }
    w[cell,1:7] <- c(X[cell,1], X[cell,2], X[cell,3], X[cell,4], X[cell,5], 
                  p[cell]*X[cell,6], (1-p[cell])*X[cell,6])
    logit(p[cell]) <- b[1] + X.FIA[cell]*b[2]
  }
  
  # Priors
  for(i in 1:2) {
    b[i] ~ dnorm(0, 0.1)
  }
  for(i in 1:7) {
    tau[i] <- 1/(sd[i]*sd[i])
    sd[i] ~ dunif(0, 2)
  }
}
